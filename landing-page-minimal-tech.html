const idp = await fetch('https://api.authmock.dev/saml/idp', {
  method: 'POST',
  headers: {
    'Authorization': 'Bearer YOUR_API_KEY',
    'Content-Type': 'application/json'
  },
  body: JSON.stringify({
    name: 'My Test IdP',
    entityId: 'https://myapp.com/saml',
    acsUrl: 'https://myapp.com/saml/acs'
  })
